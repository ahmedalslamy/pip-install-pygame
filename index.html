import pygame import random # الإعدادات الأساسية WIDTH, HEIGHT = 500, 600 GRID_SIZE = 8 # 8x8 خانات TILE_SIZE = WIDTH // GRID_SIZE COLORS = [ (255, 0, 0), # أحمر (0, 255, 0), # أخضر (0, 0, 255), # أزرق (255, 255, 0), # أصفر (255, 0, 255) # بنفسجي ] pygame.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("BRUN Match-3 Concept") # إنشاء مصفوفة اللعبة grid = [[random.choice(COLORS) for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)] def draw_grid(): for r in range(GRID_SIZE): for c in range(GRID_SIZE): pygame.draw.rect(screen, grid[r][c], (c * TILE_SIZE, r * TILE_SIZE, TILE_SIZE - 2, TILE_SIZE - 2)) def swap(pos1, pos2): r1, c1 = pos1 r2, c2 = pos2 grid[r1][c1], grid[r2][c2] = grid[r2][c2], grid[r1][c1] selected = None running = True while running: screen.fill((50, 50, 50)) draw_grid() for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.MOUSEBUTTONDOWN: x, y = pygame.mouse.get_pos() c, r = x // TILE_SIZE, y // TILE_SIZE if selected is None: selected = (r, c) else: # إذا ضغطت على خانة مجاورة، يتم التبديل if abs(selected[0] - r) + abs(selected[1] - c) == 1: swap(selected, (r, c)) selected = None pygame.display.flip() pygame.quit()
